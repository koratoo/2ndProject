
/* charge history 시퀀스 */
drop sequence ch_seq;
CREATE SEQUENCE ch_seq
       INCREMENT BY 1
       START WITH 1
       MINVALUE 1
       MAXVALUE 9999
       NOCYCLE
       NOCACHE
       NOORDER;
